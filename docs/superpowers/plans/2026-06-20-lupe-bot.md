# Lupe Bot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a WhatsApp bot that speaks as Lupe (the family dog) and sends rotating reminder messages to a family group to ensure she gets fed twice a day, stopping when someone confirms.

**Architecture:** Node.js app with five focused modules (state, messages, detector, timeUtils, scheduler) wired together by a thin index.js and a WhatsApp client wrapper. The scheduler fires cron jobs at specific times, checks state, and sends messages via the client. The client listens for group messages and runs them through the detector; confirmed messages update state and trigger a thank-you reply.

**Tech Stack:** Node.js 18, whatsapp-web.js ^1.23.0, node-cron ^3.0.3, dotenv ^16.4.5, qrcode-terminal ^0.12.0, Jest ^29.7.0 (tests)

## Global Constraints

- Timezone: `America/Santiago` — all cron jobs must use this timezone explicitly
- All messages are written in first person as Lupe the dog, in casual Chilean Spanish
- Confirmation detection is case-insensitive and accent-insensitive
- State persists to `data/state.json`; WhatsApp session persists to `data/.wwebjs_auth/`
- The `data/` directory is mounted as a Fly.io volume so it survives redeploys
- Node.js version: 18 (LTS)

```javascript
const fs = require('fs');
const path = require('path');

const TODAY = new Date().toISOString().slice(0, 10);

module.exports = function createState(filePath) {
  function load() {
    try {
      const raw = JSON.parse(fs.readFileSync(filePath, 'utf8'));
      if (raw.date !== TODAY) return { date: TODAY, morning: false, evening: false };
      return raw;
    } catch {
      return { date: TODAY, morning: false, evening: false };
    }
  }

  function save(data) {
    const dir = path.dirname(filePath);
    if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
    fs.writeFileSync(filePath, JSON.stringify(data, null, 2));
  }

  return {
    isFed(meal) {
      return load()[meal] === true;
    },
    markFed(meal) {
      const data = load();
      data[meal] = true;
      save(data);
    },
  };
};
```

```javascript
const KEYWORDS = [
  'alimentada', 'ya le di', 'yo le di', 'listo', 'ya comio', 'ya comió',
  'le di', 'ya esta', 'ya está', 'dada', 'comio', 'comió', 'alimenté', 'alimentamos',
];

const SHORT_KEYWORDS = ['ya'];

function normalize(text) {
  return text
    .toLowerCase()
    .normalize('NFD')
    .replace(/[̀-ͯ]/g, '')
    .replace(/[¡!¿?]/g, '')
    .trim();
}

function isConfirmation(text) {
  if (!text) return false;
  const norm = normalize(text);
  if (KEYWORDS.some((kw) => norm.includes(normalize(kw)))) return true;
  const wordCount = norm.split(/\s+/).filter(Boolean).length;
  if (wordCount <= 3 && SHORT_KEYWORDS.some((kw) => norm.split(/\s+/).includes(kw))) return true;
  return false;
}

module.exports = { isConfirmation };
```

```javascript
function getCurrentMeal(now = new Date()) {
  const day = now.getDay(); // 0=Sun, 1=Mon, ..., 6=Sat
  const h = now.getHours();
  const m = now.getMinutes();
  const totalMin = h * 60 + m;

  const isWeekend = day === 0 || day === 6;

  // Morning window
  const morningStart = 7 * 60 + 30;  // 07:30
  const morningEnd = isWeekend ? 13 * 60 + 30 : 13 * 60; // Sat/Sun 13:30, Mon-Fri 13:00
  if (totalMin >= morningStart && totalMin < morningEnd) return 'morning';

  // Evening window: 19:30 - 24:00
  const eveningStart = 19 * 60 + 30;
  if (totalMin >= eveningStart) return 'evening';

  return null;
}

module.exports = { getCurrentMeal };
```

```javascript
const cron = require('node-cron');
const messages = require('./messages');

const TZ = process.env.TZ || 'America/Santiago';

function schedule(expr, fn) {
  cron.schedule(expr, fn, { timezone: TZ });
}

module.exports = function setup(send, state) {
  // Helper: send reminder only if not yet fed
  function remind(meal, isFirst) {
    return async () => {
      if (state.isFed(meal)) return;
      await send(messages.getReminder(meal, isFirst));
    };
  }

  function sadAndStop(meal) {
    return async () => {
      if (state.isFed(meal)) return;
      await send(messages.getSadMessage(meal));
    };
  }

  // --- MORNING ---

  // Mon-Fri: 8:00 initial
  schedule('0 8 * * 1-5', remind('morning', true));
  // Mon-Fri: 9:30 second
  schedule('30 9 * * 1-5', remind('morning', false));
  // Mon-Fri: hourly 10:30, 11:30
  schedule('30 10 * * 1-5', remind('morning', false));
  schedule('30 11 * * 1-5', remind('morning', false));
  // Mon-Fri: 12:30 sad and stop
  schedule('30 12 * * 1-5', sadAndStop('morning'));

  // Sat-Sun: 11:00 initial
  schedule('0 11 * * 0,6', remind('morning', true));
  // Sat-Sun: 12:00 follow-up
  schedule('0 12 * * 0,6', remind('morning', false));
  // Sat-Sun: 13:00 sad and stop
  schedule('0 13 * * 0,6', sadAndStop('morning'));

  // --- EVENING (all days) ---

  // 20:00 initial
  schedule('0 20 * * *', remind('evening', true));
  // 21:00, 22:00, 23:00 hourly
  schedule('0 21 * * *', remind('evening', false));
  schedule('0 22 * * *', remind('evening', false));
  schedule('0 23 * * *', remind('evening', false));
  // 23:30 sad and stop
  schedule('30 23 * * *', sadAndStop('evening'));

  console.log('[scheduler] All cron jobs registered');
};
```

```javascript
let groupChatId = null;

async function findGroup() {
  const chats = await client.getChats();
  const group = chats.find((c) => c.isGroup && c.name === GROUP_NAME);
  if (!group) throw new Error(`Group "${GROUP_NAME}" not found. Check GROUP_NAME in .env`);
  groupChatId = group.id._serialized;
  console.log(`[client] Connected to group: ${group.name}`);

function connect(onGroupMessage) {
  client.on('ready', async () => {
    console.log('[client] WhatsApp ready');
    await findGroup();
  });

  client.on('message', async (msg) => {
    if (!groupChatId) return;
    const chat = await msg.getChat();
    if (chat.id._serialized !== groupChatId) return;
    const contact = await msg.getContact();
    const name = contact.pushname || contact.number;
    onGroupMessage(msg.body, name);
  });

  client.on('disconnected', (reason) => {
    console.error('[client] Disconnected:', reason);
    process.exit(1);
  });

  client.initialize();
}

async function sendToGroup(text) {
  if (!groupChatId) {
    console.warn('[client] Group not ready yet, skipping message');
    return;
  }
  try {
    await client.sendMessage(groupChatId, text);
  } catch (err) {
    console.error('[client] Failed to send message, retrying in 5s:', err.message);
    await new Promise((r) => setTimeout(r, 5000));
    await client.sendMessage(groupChatId, text);
  }
}

module.exports = { connect, sendToGroup };
```

```javascript
const path = require('path');
const { connect, sendToGroup } = require('./client');

const state = createState(STATE_PATH);

function onGroupMessage(body, senderName) {
  const meal = getCurrentMeal();
  if (!meal) return;
  if (state.isFed(meal)) return;
  if (!isConfirmation(body)) return;

  state.markFed(meal);
  const reply = getConfirmation(meal, senderName);
  sendToGroup(reply).catch((err) => console.error('[index] Failed to send confirmation reply:', err));
}

setupScheduler(sendToGroup, state);
connect(onGroupMessage);

console.log('[lupe-bot] Starting...');
```
